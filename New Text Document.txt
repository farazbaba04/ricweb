<script>

let freelancers = JSON.parse(localStorage.getItem("freelancers")) || [
    {name:"Waseem Akram", skill:"Full Stack Developer", premium:"React, Node.js, MongoDB", category:"Web", price:500},
    {name:"Mahnoor Islam", skill:"UI/UX Designer", premium:"Figma, Adobe XD, Branding", category:"Design", price:350},
    {name:"Sheraz Abbas", skill:"Flutter Developer", premium:"Firebase, REST APIs, App Deployment", category:"Mobile", price:400},
    {name:"Najiuallah", skill:"Backend Developer", premium:"Express.js, MySQL, APIs", category:"Web", price:450},
    {name:"Rimsha Rana", skill:"Frontend Developer", premium:"Next.js, Tailwind, SEO", category:"Web", price:420},
    {name:"Hamza Khan", skill:"Mobile App Expert", premium:"Flutter, Firebase, Play Store Deployment", category:"Mobile", price:480}
];

let selectedFreelancer = null;

function saveData(){
    localStorage.setItem("freelancers", JSON.stringify(freelancers));
}

function renderData(data){
    let container = document.getElementById("cardContainer");
    container.innerHTML="";
    data.forEach((f,i)=>{
        container.innerHTML += `
        <div class="card">
            <h3>${f.name}</h3>
            <p>${f.skill}</p>
            <small>${f.category}</small>
            <p class="price">$${f.price}</p>
            <button onclick="viewProfile(${i})">View Profile</button>
        </div>`;
    });
}

function viewProfile(index){
    selectedFreelancer = freelancers[index];
    document.getElementById("modalName").innerText = selectedFreelancer.name;
    document.getElementById("modalSkill").innerText = "Main Skill: " + selectedFreelancer.skill;
    document.getElementById("modalPremium").innerText = "Premium Skills: " + selectedFreelancer.premium;
    document.getElementById("modalCategory").innerText = "Category: " + selectedFreelancer.category;
    document.getElementById("modalPrice").innerText = "Price: $" + selectedFreelancer.price;
    document.getElementById("profileModal").style.display="flex";
}

function closeModal(){
    document.getElementById("profileModal").style.display="none";
}

function submitHire(){
    let name = document.getElementById("clientName").value;
    let email = document.getElementById("clientEmail").value;
    let details = document.getElementById("projectDetails").value;

    if(name==="" || email==="" || details===""){
        alert("Please fill all hire form fields");
        return;
    }

    alert("Hire Request Sent to " + selectedFreelancer.name + " Successfully!");
    closeModal();
}

function addFreelancer(){
    let name = document.getElementById("name").value;
    let skill = document.getElementById("skill").value;
    let premium = document.getElementById("premium").value;
    let price = document.getElementById("price").value;
    let category = document.getElementById("category").value;

    if(name==="" || skill==="" || premium==="" || price===""){
        alert("Please fill all fields");
        return;
    }

    freelancers.push({name, skill, premium, category, price});
    saveData();
    renderData(freelancers);
    alert("Freelancer Added Successfully!");
}

function filterData(){
    let search = document.getElementById("search").value.toLowerCase();
    let filter = document.getElementById("filter").value;

    let filtered = freelancers.filter(f =>
        f.name.toLowerCase().includes(search) &&
        (filter==="All" || f.category===filter)
    );

    renderData(filtered);
}

function showSection(section){
    document.getElementById("dashboard").style.display="none";
    document.getElementById("add").style.display="none";
    document.getElementById(section).style.display="block";
}

renderData(freelancers);
